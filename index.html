Fokus pada fitur esensial dan menyembunyikan kompleksitas

Contoh Implementasi OOP

<?php
// Abstract class
abstract class Kendaraan {
    protected $merk;
    protected $model;
    
    public function __construct($merk, $model) {
        $this->merk = $merk;
        $this->model = $model;
    }
    
    abstract public function nyalakan();
    
    public function getInfo() {
        return $this->merk . " " . $this->model;
    }
}

// Interface
interface GPS {
    public function navigasi($tujuan);
}

// Concrete class
class Mobil extends Kendaraan implements GPS {
    private $jumlahPintu;
    
    public function __construct($merk, $model, $jumlahPintu) {
        parent::__construct($merk, $model);
        $this->jumlahPintu = $jumlahPintu;
    }
    
    public function nyalakan() {
        return "Mobil " . $this->getInfo() . " dinyalakan dengan kunci";
    }
    
    public function navigasi($tujuan) {
        return "Navigasi ke " . $tujuan . " dimulai";
    }
}

// Penggunaan
$mobil = new Mobil("Toyota", "Camry", 4);
echo $mobil->nyalakan() . "\n";
echo $mobil->navigasi("Jakarta");
?>

Database Model Class

class User {
    private $pdo;
    
    public function __construct(PDO $pdo) {
        $this->pdo = $pdo;
    }
    
    public function create($name, $email) {
        $stmt = $this->pdo->prepare(
            "INSERT INTO users (name, email, created_at) VALUES (?, ?, NOW())"
        );
        return $stmt->execute([$name, $email]);
    }
    
    public function findById($id) {
        $stmt = $this->pdo->prepare("SELECT * FROM users WHERE id = ?");
        $stmt->execute([$id]);
        return $stmt->fetch();
    }
    
    public function findAll($limit = 10, $offset = 0) {
        $stmt = $this->pdo->prepare(
            "SELECT * FROM users ORDER BY created_at DESC LIMIT ? OFFSET ?"
        );
        $stmt->execute([$limit, $offset]);
        return $stmt->fetchAll();
    }
    
    public function update($id, $name, $email) {
        $stmt = $this->pdo->prepare(
            "UPDATE users SET name = ?, email = ?, updated_at = NOW() WHERE id = ?"
        );
        return $stmt->execute([$name, $email, $id]);
    }
    
    public function delete($id) {
        $stmt = $this->pdo->prepare("DELETE FROM users WHERE id = ?");
        return $stmt->execute([$id]);
    }
}

Keamanan Web: Membangun Benteng Digital

Keamanan adalah aspek kritis dalam pengembangan web. Pelajari ancaman umum dan cara mengatasinya.

üíâ

SQL Injection

Penyerang menyuntikkan kode SQL berbahaya melalui input user

‚ùå Berbahaya:

"SELECT * FROM users WHERE id = " . $_GET['id']

‚úÖ Aman:

$stmt->execute([$_GET['id']])

üé≠

Cross-Site Scripting (XSS)

Penyerang menyuntikkan script JavaScript ke halaman web

‚ùå Berbahaya:

echo $_POST['comment']

‚úÖ Aman:

echo htmlspecialchars($_POST['comment'])

üîê

Password Hashing

Jangan pernah simpan password dalam bentuk plain text

‚ùå Berbahaya:

md5($password)

‚úÖ Aman:

password_hash($password, PASSWORD_DEFAULT)

Implementasi Keamanan Komprehensif

class SecurityHelper {
    
    // CSRF Protection
    public static function generateCSRFToken() {
        if (empty($_SESSION['csrf_token'])) {
            $_SESSION['csrf_token'] = bin2hex(random_bytes(32));
        }
        return $_SESSION['csrf_token'];
    }
    
    public static function verifyCSRFToken($token) {
        return isset($_SESSION['csrf_token']) && 
               hash_equals($_SESSION['csrf_token'], $token);
    }
    
    // Input Sanitization
    public static function sanitizeInput($data) {
        $data = trim($data);
        $data = stripslashes($data);
        $data = htmlspecialchars($data, ENT_QUOTES, 'UTF-8');
        return $data;
    }
    
    // Password Security
    public static function hashPassword($password) {
        return password_hash($password, PASSWORD_ARGON2ID, [
            'memory_cost' => 65536,
            'time_cost' => 4,
            'threads' => 3
        ]);
    }
    
    public static function verifyPassword($password, $hash) {
        return password_verify($password, $hash);
    }
    
    // File Upload Security
    public static function validateFileUpload($file, $allowedTypes = [], $maxSize = 2097152) {
        if ($file['error'] !== UPLOAD_ERR_OK) {
            return false;
        }
        
        if ($file['size'] > $maxSize) {
            return false;
        }
        
        $finfo = finfo_open(FILEINFO_MIME_TYPE);
        $mimeType = finfo_file($finfo, $file['tmp_name']);
        finfo_close($finfo);
        
        return in_array($mimeType, $allowedTypes);
    }
}

// RESTful API dengan routing
class ApiController {
    
    public function index(Request $request) {
        $limit = $request->get('limit', 10);
        $users = User::paginate($limit);
        
        return response()->json([
            'data' => $users->items(),
            'meta' => [
                'total' => $users->total(),
                'per_page' => $users->perPage(),
                'current_page' => $users->currentPage()
            ]
        ]);
    }
    
    public function store(Request $request) {
        $validator = validator($request->all(), [
            'name' => 'required|string|max:255',
            'email' => 'required|email|unique:users'
        ]);
        
        if ($validator->fails()) {
            return response()->json([
                'error' => 'Validation failed',
                'messages' => $validator->errors()
            ], 422);
        }
        
        $user = User::create($validator->validated());
        
        return response()->json($user, 201);
    }
}

Queue & Background Jobs

// Job untuk mengirim email
class SendEmailJob {
    protected $user;
    protected $message;
    
    public function __construct($user, $message) {
        $this->user = $user;
        $this->message = $message;
    }
    
    public function handle() {
        // Kirim email menggunakan library email
        $mailer = new Mailer();
        $mailer->send(
            $this->user->email,
            'Subject',
            $this->message
        );
        
        log_info("Email sent to {$this->user->email}");
    }
    
    public function failed(Exception $exception) {
        log_error("Failed to send email: " . $exception->getMessage());
    }
}

// Dispatch job
$job = new SendEmailJob($user, 'Welcome!');
Queue::push($job);

Testing dengan PHPUnit

use PHPUnit\Framework\TestCase;

class UserTest extends TestCase {
    
    private $user;
    
    protected function setUp(): void {
        $this->user = new User();
    }
    
    public function testUserCanBeCreated() {
        $userData = [
            'name' => 'John Doe',
            'email' => 'john@example.com'
        ];
        
        $user = User::create($userData);
        
        $this->assertInstanceOf(User::class, $user);
        $this->assertEquals('John Doe', $user->name);
        $this->assertEquals('john@example.com', $user->email);
    }
    
    public function testUserValidation() {
        $this->expectException(ValidationException::class);
        
        User::create([
            'name' => '', // Invalid: empty name
            'email' => 'invalid-email' // Invalid: bad email format
        ]);
    }
    
    public function testApiEndpoint() {
        $response = $this->json('GET', '/api/users');
        
        $response->assertStatus(200)
                 ->assertJsonStructure([
                     'data' => [
                         '*' => ['id', 'name', 'email']
                     ],
                     'meta'
                 ]);
    }
}
